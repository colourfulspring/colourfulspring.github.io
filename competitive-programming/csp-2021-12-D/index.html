    2021.12 ccf csp 第四题 磁盘文件操作

    法一

    思路

    注意到每个磁盘位置最多只能被一个进程占有，可以为每个磁盘区间构造一个结构体，包含三个成员：区间左端点\(l\)，该区间被进程占有情况（\(id=0\)表示没被占有，\(id>0\)表示被第\(id\)号进程占有，\(id<0\)表示没被占有，但上次被占有是被第\(-id\)号进程占有），该区间存的值。不需要维护右端点\(r\)，使用后继的左端点表示即可。

    若以区间左端点\(l\)为排序的键，则使用\(std::map\)可以实现一个支持题中操作的数据结构。将除去键\(l\)以外的\(id\)，\(val\)结合，定义为结构体\(info\)，详细定义见代码。

    算法

    初始情况下，\(map\)内有两个元素：\(\{1,info(0,0)\}\)和\(\{m+1,info(0,0)\}\)。前者表示\(1\)开始的所有磁盘位置均存\(0\)，且未被进程占有。后者表示哨兵，暗示前者的区间右端点\(m\)。总之，二者表示\([1,m]\)初始化为\(0\)，且无进程占有。每个操作的详细过程如下：

    • \(0\ id\ l\ r\ x\)：找到\(map\)中不大于\(l\)的，且键最大的节点\(start\)。从\(start\)开始，按序枚举节点直到节点键值大于\(r\)结束，设枚举到节点\(end\)结束。每个节点如果能被写入，则将其\(info\)属性改为写入后的值\(\{id,x\}\)。有两个注意点：

      • 若\([l,r]\)不是\(start\)和\(end\)两个节点的键，我们需要插入新的以\(l\)，\(r\)为键的节点。
      • \(start\)和\(end\)之间所有能被写入的节点，写入完成之后结果一样，可以直接合并到以\(l\)开头的那个节点上去。

    • \(1\ id\ l\ r\)， \(2\ id\ l\ r\)：找到\(map\)中不大于\(l\)的，且键最大的节点\(start\)。从\(start\)开始，按序枚举节点直到节点键值大于\(r\)结束，设枚举到节点\(end\)时结束。检查每个节点\(id\)是否符合操作的条件。如果符合，再枚举一次节点\(start\)到节点\(end\)，对每个节点\(id\)值进行修改。有两个注意点：

      • 在判断的时候不用管\([l,r]\)是不是\(start\)和\(end\)两个节点的键。但在修改的时候，若\([l,r]\)不是\(start\)和\(end\)两个节点的键，我们需要插入新的以\(l\)，\(r\)为键的节点。
      • 不需要管合并节点的问题。因为所有能被改的节点\(id\)都一样，它们没有合并是因为\(val\)不一样。\(val\)不会在\(1\)，\(2\)两种操作被改掉，所以不用管。

    • \(3\ p\)：找到\(map\)中不大于\(p\)的，且键最大的节点\(start\)。若\(start\)的\(id>0\)，则输出\(start\)的\(id\) \(val\)，若\(start\)的\(id<=0\)，则输出\(0\) \(0\)。

    代码

    #include <bits/stdc++.h>
using namespace std;

struct info {
    int id, val;
};

int n, m, k;
map<int, info> tree;

int write(int id, int l, int r, int x) {
    auto it = --tree.upper_bound(l);

    if (it->second.id > 0 && it->second.id != id)
        return 0;
    if (it->first < l) {
        tree.emplace(l, it->second);
        ++it;
    }
    int ans = 0;
    auto p = it;
    while (true) {
        if (r <= next(p)->first) {
            if (r < next(p)->first)
                tree.emplace(r, p->second);
            if (p != it)
                tree.erase(p++);
            else
                it->second = {id, x};
            ans = r;
            break;
        }
        if (p != it)
            tree.erase(p++);
        else {
            ++p;
            it->second = {id, x};
        }
        if (p->second.id > 0 && p->second.id != id) {
            ans = p->first;
            break;
        }
    }
    return ans;
}

bool editID(int id, int l, int r, int op) {
    auto it = --tree.upper_bound(l);

    int newID = op ? id : -id;
    auto p = it;
    bool ans = true;
    while (true) {
        if (op && p->second.id != -id || !op && p->second.id != id) {
            ans = false;
            break;
        }
        if (r <= next(p)->first) {
            break;
        }
        ++p;
    }
    if (!ans)
        return false;

    if (it->first < l) {
        tree.emplace(l, it->second);
        ++it;
    }
    p = it;
    while (true) {
        if (r <= next(p)->first) {
            if (r < next(p)->first)
                tree.emplace(r, p->second);
            p->second.id = newID;
            break;
        }
        p->second.id = newID;
        ++p;
    }
    return true;
}

pair<int, int> query(int p) {
    auto it = --tree.upper_bound(p);
    if (it->second.id > 0)
        return make_pair(it->second.id, it->second.val);
    else
        return make_pair(0, 0);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;
    int type = 0, id = 0, l = 0, r = 0, x = 0, p = 0;
    tree.emplace(1, info{0, 0});
    tree.emplace(m + 1, info{0, 0});
    for (int i = 0; i < k; ++i) {
        cin >> type;
        switch (type) {
        case 0:
            cin >> id >> l >> r >> x;
            cout << write(id, l, r + 1, x) - 1 << endl;
            break;
        case 1:
            cin >> id >> l >> r;
            cout << (editID(id, l, r + 1, 0) ? "OK" : "FAIL") << endl;
            break;
        case 2:
            cin >> id >> l >> r;
            cout << (editID(id, l, r + 1, 1) ? "OK" : "FAIL") << endl;
            break;
        default:
            cin >> p;
            auto ans = query(p);
            cout << ans.first << ' ' << ans.second << endl;
        }
    }
    return 0;
}
